*define
;CSVファイル入出力のサンプルスクリプトです。
game
*start
select "書き込み実験",*write,"読み込み実験",*read

*write
csvopen "test.csv","w" ;ここで二つ目を"wc"にすれば暗号化します。
csvwrite "テスト","ですよ",1
mov $0,"変数も指定"
mov $1,"できますよ"
mov %0,2
csvwrite $0,$1,%0
csvwrite "終わりです","終わりだよ",3
csvclose
end

*read
fileexist %0,"test.csv"
if %0=0 end
;ファイルがあるかどうか分からない場合は、このように事前にファイルの有無チェックをしたほうがいいでしょう。
;必ずある場合はやらなくても構いません。

csvopen "test.csv","r"
;上で作ったファイルを読み込みます。
*loop
csveof %0
if %0=1 csvclose:end ;終了を検出したら終わります。
csvread $0,$1,%0
$0，$1，%0@
goto *loop
